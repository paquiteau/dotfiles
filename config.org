(defun tangle-remote (name &optional host sudo)
  (cond ((file-remote-p (buffer-file-name))
         (let* ((tramp-con (tramp-dissect-file-name (buffer-file-name)))
                (tramp-home (tramp-get-home-directory tramp-con))
                (tramp-host (tramp-file-name-host tramp-con)))
           (when (or (not host) (string= tramp-host host))
             (expand-file-name (format "/sshx:%s:%s/%s" tramp-host tramp-home name)))))
        ((or (not host) (string= host system-name)) (expand-file-name name (getenv "HOME")))
        (t nil))
  )

# Formats two VCS_info messages, one with colors, and one without
vcs_formats(){
    local action=$1 branch=$2 base=$3 rev=$4
    local msg
    local -i i
    # with color support
    msg="[${nocolor_prompt}${vcs_repo_color}%s${vcs_sep_color}/${vcs_branch_color}%s"
    msg=$(printf $msg ${base/*\/} $branch)
    if [[ -n ${rev} ]] ; then
        msg="${msg}${vcs_colon_color}:${vcs_rev_color}${rev}"
    fi
    if [[ -n ${action} ]] ; then
        msg="${msg}${nocolor_prompt}|${vcs_action_color}${action}"
    fi
    VCS_info="${msg}${linecolor}]"
    return 0
}

# Git is powerfull
git_detect(){
    git rev-parse --is-inside-work-tree &> /dev/null
}

# Find git's branch
git_getbranch (){
    local gitbranch gitdir=$1 tmp actiondir
    local gitsymref='git symbolic-ref HEAD'

    # In certain circumstances, we have to take into account
    # actions
    actiondir=''
    for tmp in "${gitdir}/rebase-apply" \
               "${gitdir}/rebase"       \
               "${gitdir}/../.dotest"; do
        if [[ -d ${tmp} ]]; then
            actiondir=${tmp}
            break
        fi
    done
    if [[ -n ${actiondir} ]]; then
        gitbranch="$(${gitsymref} 2> /dev/null)"
        [[ -z ${gitbranch} ]] && [[ -r ${actiondir}/head-name ]] \
            && gitbranch="$(< "${actiondir}"/head-name)"

    # MERGE_HEAD state
    elif [[ -f "${gitdir}/MERGE_HEAD" ]] ; then
        gitbranch="$(eval $gitsymref 2> /dev/null)"
        [[ -z ${gitbranch} ]] && gitbranch="$(< ${gitdir}/MERGE_HEAD)"

    # rebase
    elif [[ -d "${gitdir}/rebase-merge" ]] ; then
        gitbranch="$(< ${gitdir}/rebase-merge/head-name)"

    # dotest
    elif [[ -d "${gitdir}/.dotest-merge" ]] ; then
        gitbranch="$(< ${gitdir}/.dotest-merge/head-name)"

    # Normal case
    else
        gitbranch="$(eval $gitsymref 2> /dev/null)"

        # shit happens
        if [[ $? -ne 0 ]] ; then
            gitbranch="refs/tags/$(git describe --exact-match HEAD 2>/dev/null)"

            # big shit happens
            if [[ $? -ne 0 ]] ; then
                gitbranch=$(< $gitdir/HEAD)
                gitbranch="${gitbranch:0:7}..."
            fi
        fi
    fi

    # keep only the last part of gitbranch
    printf '%s' "${gitbranch#refs/[^/]*/}"
    return 0
}

git_getaction(){
    local gitaction='' gitdir=$1
    local tmp

    for tmp in "${gitdir}/rebase-apply" \
               "${gitdir}/rebase"       \
               "${gitdir}/../.dotest" ; do
        if [[ -d ${tmp} ]] ; then
            if   [[ -f "${tmp}/rebasing" ]] ; then
                gitaction="rebase"
            elif [[ -f "${tmp}/applying" ]] ; then
                gitaction="am"
            else
                gitaction="am/rebase"
            fi
            printf '%s' ${gitaction}
            return 0
        fi
    done

    for tmp in "${gitdir}/rebase-merge/interactive" \
               "${gitdir}/.dotest-merge/interactive" ; do
        if [[ -f "${tmp}" ]] ; then
            printf '%s' "rebase-i"
            return 0
        fi
    done

    for tmp in "${gitdir}/rebase-merge" \
               "${gitdir}/.dotest-merge" ; do
        if [[ -d "${tmp}" ]] ; then
            printf '%s' "rebase-m"
            return 0
        fi
    done

    if [[ -f "${gitdir}/MERGE_HEAD" ]] ; then
        printf '%s' "merge"
        return 0
    fi

    if [[ -f "${gitdir}/BISECT_LOG" ]] ; then
        printf '%s' "bisect"
        return 0
    fi
    return 1
}

git_get_data(){
    local gitdir gitbase gitbranch gitaction

    gitdir=".git"
    gitbranch="$(git_getbranch ${gitdir})"

    if [[ -z ${gitdir} ]] || [[ -z ${gitbranch} ]] ; then
        return 1
    fi

    gitaction="$(git_getaction ${gitdir})"
    gitprefix=$(git rev-parse --show-prefix)
    gitbase=${PWD%/${gitprefix%/}}
    vcs_formats "${gitaction}" "${gitbranch}" "${gitbase}" ''
    return 0
}

COLUMNS=$(tput cols)
# Génération de la ligne de "-"
function prompt_command
{
    local pwd ERR DATE PROMPT DIR POST_DIR
    # Cette fonction génère un prompt sur deux lignes
    # ayant cette allure :
    # ─( 12:59:05 )─< ~ >───────────────────────────────────────────────────────[ 0 ]─
    # user@host $

    # À décommenter si on veut afficher des infos
    # quand on se trouve dans un dépôt versionné

    local -ax msg
    git_detect && git_get_data || VCS_info=""
    # Chemin courant, en faisant attention à la largeur de la fenêtre
    pwd=${PWD/#$HOME/'~'}
    if (( ${#pwd} + ${VCS_size} + 27 > COLUMNS )); then
 	    if (( COLUMNS >= 33 )); then
	        DIR='..'${pwd:${#pwd}+${VCS_size}+30-COLUMNS}
            POST_DIR='${MINUS_LINE:0:4-${#?}}'
	    else
	        DIR=$pwd
            POST_DIR=$MINUS_CHAR
	    fi
    else
	    DIR=$pwd
        POST_DIR='${MINUS_LINE:'${#pwd}+${VCS_size}+1'+${#?}}'
    fi

    if [[ $VIRTUAL_ENV != "" ]];then
        VENV="(p:${VIRTUAL_ENV##*/})"
    elif [[ $CONDA_PROMPT_MODIFIER != "" ]]; then
         VENV="(c:${CONDA_PROMPT_MODIFIER:1:-2})"
    else
        VENV=''
    fi
    if [[ ${STY} != "" ]]
    then
        SCREEN_NAME=${STY//[[:digit:]]/}
        SCREEN_ENV="[${SCREEN_NAME#.}]"
    fi

    # On affiche le code de retour en "no color" si c'est 0, en rouge sinon
    # Ceci n'est pas customizable à coup de variable,
    # il faut changer $((31+($?==0)*6)) si vous y tenez vraiment et que vous comprenez ce que vous faites
    ERR=$'[ \[\033[1;$((31+($?==0)*6))m\]$?'${linecolor}' ]'
    # Affichage de l'heure courante
    # pratique pour savoir quand on a tapé la commande précédente

    # Deuxième ligne du prompt "user@host $"
    # Attention "\\\$" devient \$, c'est-à-dire # pour root, $ pour les autres
if [ -x /usr/bin/tput ] && tput setaf 1 >&/dev/null; then
    color_prompt=yes
    PROMPT="${username_color_prompt}\u${symbols_color_prompt}@${host_color_prompt}\h${symbols_color_prompt}${SCREEN_ENV}${VENV}${symbols_color_prompt}\\\$ ${nocolor_prompt}"
    # On fusionne tout ça
    PS1="${linecolor}─( ${date_color}\D{%H:%M:%S}${linecolor} )─< ${pwd_color}$DIR${linecolor} >─${VCS_info}─$ERR─\n${PROMPT}"
else
    color_prompt=no
    PS1="\u@\h\\\$"
fi
}

# On régénére le prompt après chaque commande
PROMPT_COMMAND=prompt_command

DOMAINS=("pierre-antoine-comby.fr" "imap.ens-paris-saclay.fr" "smtp.ens-paris-saclay.fr")

# 1. Detect the local gateway IP and the exact local Interface Devic
# Looks for a default route line that does NOT contain VPN keywords
GW_LINE=$(ip route | grep default | grep -v -E '(tun|ppp|fct|vpn|enp|lo)' | head -n 1)

LOCAL_GW=$(echo "$GW_LINE" | awk '{print $3}')
LOCAL_DEV=$(echo "$GW_LINE" | awk '{print $5}')

if [ -z "$LOCAL_GW" ] || [ -z "$LOCAL_DEV" ]; then
    echo "❌ Error: Could not detect your local gateway or interface network device."
    exit 1
fi

echo "📌 Local gateway detected: $LOCAL_GW on device: $LOCAL_DEV"
echo "----------------------------------------"

# 2. Loop through each domain, resolve its IP, and add the strict bypass route
for DOMAIN in "${DOMAINS[@]}"; do
    echo "Resolving $DOMAIN..."
    
    IPS=$(dig +short "$DOMAIN" | grep -E '^[0-9.]+$')
    
    if [ -z "$IPS" ]; then
        echo "⚠️  Warning: Could not resolve $DOMAIN"
        continue
    fi
    
    for IP in $IPS; do
        echo "✈️  Routing $IP via $LOCAL_GW on $LOCAL_DEV (Bypassing VPN)..."
        
        # Force the rule to use the explicit local physical network device interface
        sudo ip route add "$IP" via "$LOCAL_GW" dev "$LOCAL_DEV" 2>/dev/null
        
        if [ $? -eq 0 ]; then
            echo "✅ Successfully added route for $IP"
        else
            echo "ℹ️  Route for $IP already exists or failed to apply."
        fi
    done
done
echo "----------------------------------------"
echo "🎉 Split tunneling strictly applied!"
